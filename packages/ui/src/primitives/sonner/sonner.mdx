# Toast

A succinct message that is displayed temporarily.

## Preview

<Canvas of={ToastStories.Default} />

## Usage

### Add `Toaster` component to your layout

export const providerCode = `import { Toaster } from "@hooper/ui/sonner";

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <head />
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  );
}`;

<Source code={providerCode} language="tsx" dark />

### Use `sonner` function from

export const importCode = `import { toast } from "sonner";`;

export const usageCode = `export const ToastDemo = () => {

  return (
    <Button
      onClick={() => {
        toast("Hello, world!");
      }}
    >
      Show Toast
    </Button>
  );
}
`;

<Source code={importCode} language="ts" dark />
<Source code={usageCode} language="tsx" dark />
